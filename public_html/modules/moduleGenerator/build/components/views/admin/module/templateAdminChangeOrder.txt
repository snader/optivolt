$sSortableJavascript = <<<EOT
<script>
    $('ol.sortable').sortable({
        axis: 'y',
        forcePlaceholderSize: true,
        helper: 'clone',
        items: 'li',
        cancel: '.not-sortable',
        opacity: .6,
        placeholder: 'placeholder',
        revert: 250,
        tolerance: 'pointer',
        stop: function(){
            makeZebra('#itemSorter');
        }
        }).disableSelection();

    function setOrder(){
        var order = '';
        $('#itemSorter li').each(function(index, element){
            order += (order == '' ? '' : '|')+$(element).data('itemid');
        });
        $("#order").val(order);
        return true;
    }

    makeZebra('#itemSorter');
</script>
EOT;
